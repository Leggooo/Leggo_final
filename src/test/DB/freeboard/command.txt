 create table freeboard(
  free_brd_type_num number(7) primary key,
  free_brd_num varchar2(60) not null,
  title varchar2(60) not null,
  content varchar2(1000) not null,
  update_date date,
  user_id varchar2(20) not null,
  img_path varchar2(60)
  );
  
 create table img(
  brd_type_num number(7) primary key,
  brd_num varchar2(60) not null,
  content_url varchar2(1000) not null
  );
  
 create table cmt(
  free_brd_type_num number(7) references freeboard (free_brd_type_num),
  cmt_num number(7) primary key,
  cmt_content varchar2(500) not null,
  update_date date,
  user_id varchar2(20) references member (user_id)
  );
